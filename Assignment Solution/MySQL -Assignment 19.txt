1) Create a view that shows all of the customers who have the highest ratings.

-> CREATE VIEW h_rated_cust AS
SELECT *
FROM customers
WHERE rating=
    (SELECT max(rating)
     FROM customers);

2) Create a view that shows the number of salespeople in each city.

-> SELECT *
FROM h_rated_cust;


CREATE VIEW tns_city AS
SELECT city,
       count(city) "NO. OF SALES PEOPLE"
FROM salespeople
GROUP BY city;

 --View created.

SELECT *
FROM tns_city;

3)  Create a view that shows the average and total orders for each salesperson after
his or her name. Assume all names are unique.

-> CREATE VIEW AV_TOT_SAL AS
SELECT sname,
       avg(onum) AVERAGE,
       count(onum) "TOTAL ORDERS"
FROM salespeople
JOIN orders using(snum)
GROUP BY sname;

4) Create a view that shows each salesperson with multiple customers.

-> CREATE VIEW m_c AS
SELECT sname,
       cname
FROM salespeople
JOIN customers USING (snum);
